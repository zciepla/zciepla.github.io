Skrypt zmieniający nazwę - dopisuje sufiks postaci "_old"

#!/bin/bash
for nazwa in `ls`
 do
   echo "$0"
   echo "$nazwa"
        if [ $nazwa != `basename $0` ]
        then
        cp $nazwa $nazwa"_old"
        fi
 done
exit 0
